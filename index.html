  <script>

    // Initialize TimelineJS with your Google Sheet

    document.addEventListener('DOMContentLoaded', function () {

      const options = {

        font: 'Default',

        lang: 'en',

        timenav_position: 'top',

        initial_zoom: 2,

        start_at_slide: 0

      };


      const sheetUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vTASeBqotFAqXDWlXw05dNDAOVrUIcWdzSvFkWOE5GLDh4uqrNyZXxALwbgLf4x2uzI42j5FnIeW0UR/pubhtml';


      const timeline = new TL.Timeline('timeline-embed', sheetUrl, options);


      // Post our height to parent for auto-resizing if the parent listens

      function postHeight() {

        const h = document.documentElement.scrollHeight || document.body.scrollHeight || 0;

        window.parent && window.parent.postMessage({ height: h }, '*');

      }


      // After initial render and on resize/slide changes

      timeline.on('loaded', postHeight);

      window.addEventListener('resize', postHeight);

      // A small periodic check for a moment, to catch late layout changes (images etc.)

      let checks = 0;

      const interval = setInterval(() => {

        postHeight();

        if (++checks > 20) clearInterval(interval);

      }, 250);

    });

  </script>
